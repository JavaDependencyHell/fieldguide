% --- Section spacing (A5 tuned) via KOMA-Script native commands ---
% We use \RedeclareSectionCommand because titlesec is incompatible with scrbook/KOMA-Script used in this project.

\RedeclareSectionCommand[
  beforeskip=2.2ex,
  afterskip=0.5ex
]{section}

\RedeclareSectionCommand[
  beforeskip=1.8ex,
  afterskip=0.5ex
]{subsection}

\RedeclareSectionCommand[
  beforeskip=1.4ex,
  afterskip=0.35ex
]{subsubsection}

% --- Paragraph density ---
\setlength{\parskip}{0.35em}
\setlength{\parindent}{0pt}

% --- Display Math spacing ---
\setlength{\abovedisplayskip}{0.7em}
\setlength{\belowdisplayskip}{0.7em}

% --- Float breathing (A5 needs tighter control) ---
\setlength{\textfloatsep}{0.8em}
\setlength{\floatsep}{0.7em}
\setlength{\intextsep}{0.7em}

% --- Lists tighter (important for field guides) ---
\usepackage{enumitem}
\setlist{
  itemsep=0.2em,
  topsep=0.4em,
  parsep=0em,
  partopsep=0em
}
