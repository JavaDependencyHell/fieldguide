\usepackage{etoolbox}

% --------------------------------------------------
% Paragraph baseline (format-neutral)
% --------------------------------------------------

\setlength{\parindent}{0pt}

% NOTE: parskip is format-tuned in size-specific files

% --------------------------------------------------
% Section styling (compact but readable)
% --------------------------------------------------

\RedeclareSectionCommand[
  beforeskip=1.6ex,
  afterskip=0.8ex,
  afterindent=false,
  font=\normalfont\Large\bfseries,
]{section}

% Thin rule above sections

\pretocmd{\section}{\rule{\linewidth}{0.4pt}\vspace{0.3ex}}{}{}

% --------------------------------------------------
% Subsection — RUN-IN (primary density win)
% --------------------------------------------------

\RedeclareSectionCommand[
  runin=true,
  afterindent=false,
  beforeskip=1.7ex,
  afterskip=0.74em,
  font=\normalfont\bfseries,
]{subsection}

% --------------------------------------------------
% Subsubsection — stepped hierarchy
% --------------------------------------------------

\RedeclareSectionCommand[
  runin=true,
  afterindent=false,
  beforeskip=1.1ex,
  afterskip=0.75em,
  font=\normalfont\bfseries\scshape,
]{subsubsection}

% Lists — field guide density
% --------------------------------------------------

  topsep=0.4em,
  parsep=0em,
  partopsep=0em
